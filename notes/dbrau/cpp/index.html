    cout<< "Roll number: "<<roll_number<<\n;

     sub1 = x;
     sub2 = y;

     cout << "Marks in Sub1: " <<sub1<<\n;
     cout << "Marks in Sub2: " <<sub2<<\n;
 }

 {
     float total;
     public: void display();
 };
 void result::display()

     total = sub1 + sub2;
