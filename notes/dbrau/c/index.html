   printf("Enter array element of a[%d]:",i);

  }

  {
    mid=(lb+ub)/2;
    if(arr[mid]==n)
    {
       printf("Element found ");
       break;

    if(arr[mid]>n)
       ub=mid-1;
    if(arr[mid]<n)
       lb=mid+1;
  }

  {
    printf("Element not found");
  }
 }

Solution:

#include<conio.h>
void main()
{
	int i,j,t,n;
	printf("Enter terms: ");
	scanf("%d",&n);
	int a[n];
	printf("enter array elements:\n");
	for (i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	 for (i=0; i<n;i++)
	 {
	 	for (j=0;j<n-i;j++)
	 	{
	 		if (a[j]<a[j+1])
	 		{
	 			t=a[j];
	 			a[j]=a[j+1];
	 			a[j+1]=t;
	 		}
	 	}
	 }
	 printf("sorted array is:\n");
	 for (i=0;i<n;i++)
	 {
	 	printf("%d\n",a[i]);
